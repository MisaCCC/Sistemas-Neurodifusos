Función SMF
Esta función calcula valores de pertenencia difusos utilizando una función de pertenencia en forma de S basada en spline. También puede calcular esta función de membresía utilizando un objeto fismf.
y=smf(x,params) devuelve valores de pertenencia difusos calculados utilizando la función de pertenencia en forma de S basada en spline dada por:
Función PSIGMF
Esta función calcula valores de pertenencia difusos utilizando el producto de dos funciones de pertenencia sigmoidal. También puede calcular esta función de membresía utilizando un objeto fismf.
y=psigmf(x,params) devuelve valores de pertenencia difusos calculados mediante el producto de dos funciones de pertenencia sigmoideas.

P4. Función creada

function C=compDif(x,A,tipo)
titulo="";
switch tipo(1)
case 1 %Estandar
C=1-A;
titulo="Complemento estándar";
case 2 %Sugeno
if(tipo(2)>-1)
C=(1-A)./(1+tipo(2).*A);
else
C=1-A;
end
titulo="Complemento Sugeno";
case 3 %Yager
if(tipo(2)>0)
C=(1-A.^tipo(2)).^(1./tipo(2));
else
C=(1-A);
end
titulo="Complemento Yager";
end
subplot(2,1,1)
plot(x,A)
title("Conjunto difuso")
ylim([0 1])
subplot(2,1,2)
plot(x,C)
title(titulo)
ylim([0 1])
end

P7. Función creada

function C=interDifd(x,A,B,tipo)
titulo="";
switch tipo(1)
case 1 %Minimo
C=min(A,B);
titulo="Intersección (minimo)";
case 2 %Producto algebráico
C=A.*B;
titulo="Intersección (producto algebraico)";
case 3 %Producto acotado
C=max(0,A+B-1);
titulo="Intersección (producto acotado)";
end
subplot(3,1,1)
gcdfd(x,A)
title("Conjunto difuso")
ylim([0 1])
subplot(3,1,2)
gcdfd(x,B)
title("Conjunto difuso")
ylim([0 1])
subplot(3,1,3)
gcdfd(x,C)
title(titulo)
ylim([0 1])
end

P4. Función de membresía de A

function C=uA(x)
c1=x<=1.7;
c2=(1.7<=x & x<=2);
c3=(2<=x & x<=2.2);
c4=x>=2.2;
C(c1)=0;
C(c2)=(x(c2)-1.7)/(2-1.7);
C(c3)=(2.2-x(c3))/(2.2-2);
C(c4)=0;
end

P5. Función de membresía de B

function C=uB(x)
c1=x<=-2;
c2=(-2<=x & x<=1);
c3=(1<=x & x<=2);
c4=x>=2;
C(c1)=0;
C(c2)=(x(c2)-(-2))/(1-(-2));
C(c3)=(2-x(c3))/(2-1);
C(c4)=0;
end
