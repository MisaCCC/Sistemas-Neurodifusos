if(!mask) { # if the user does not want print out statements

}
g = gradient(x) # find gradient at x

}
if(norm(g, "F") < tol) { # convert gradient into single value
# frobenius
print("Stationary Point Found at point:")
print(x)

return(x)

p = -g
alpha = -(t(g)%*%p) / (t(p)%*%hessian(x)%*%p)
